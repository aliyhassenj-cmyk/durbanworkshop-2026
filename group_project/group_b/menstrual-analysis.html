Group B — Menstrual Cycle Analysis

Cytokines, Immune Cells, and Birth Control

This document explores the Group B (menstrual cycle) dataset. We’ll:

  1. Load and join the data
  2. Make summary tables with p-values
  3. Visualise cytokine and immune cell differences by birth control and time point
  4. Run regression models

1. Load and Join Data

# --- Sample IDs ---
sample_ids <- read_csv(file.path(data_dir, "group_b_menstruation",
                                  "00_sample_ids_period.csv"),
                       show_col_types = FALSE)

# --- Participant metadata ---
metadata <- read_csv(file.path(data_dir, "group_b_menstruation",
                                "01_participant_metadata_period.csv"),
                     show_col_types = FALSE)

# --- Luminex cytokines (long format) ---
luminex <- read_csv(file.path(data_dir, "group_b_menstruation",
                               "02_luminex_period.csv"),
                    show_col_types = FALSE)

# --- Flow cytometry ---
flow <- read_csv(file.path(data_dir, "group_b_menstruation",
                            "03_flow_cytometry_period.csv"),
                 show_col_types = FALSE)

Join everything together

# Join sample IDs with metadata (keep only .x versions, rename)
samples <- sample_ids %>%
  left_join(metadata, by = "pid", relationship = "many-to-one") %>%
  rename(arm = arm.x) %>%
  select(-arm.y)

# Widen cytokines (one column per cytokine)
# Clean names: replace hyphens with underscores in ALL column names
luminex_wide <- luminex %>%
  filter(limits == "not_censored") %>%
  select(sample_id, cytokine, conc) %>%
  pivot_wider(names_from = cytokine, values_from = conc)

# Replace hyphens in column names with underscores
names(luminex_wide) <- str_replace_all(names(luminex_wide), "-", "_")

cyt_names <- setdiff(names(luminex_wide), "sample_id")

# Join cytokine values onto samples
data_cyt <- samples %>%
  left_join(luminex_wide, by = "sample_id")

# Join flow cytometry
data_full <- data_cyt %>%
  left_join(flow, by = "sample_id")

2. Summary Tables with P-values

2.1 Cytokine levels by birth control arm

# Helper: run a t-test for each cytokine and return a tidy table
cyt_arm <- data_cyt %>%
  pivot_longer(cols = all_of(cyt_names),
               names_to = "cytokine", values_to = "conc") %>%
  drop_na(conc, arm)

cyt_arm %>%
  group_by(cytokine, arm) %>%
  summarise(n = n(),
            mean_conc = mean(conc, na.rm = TRUE),
            sd_conc   = sd(conc, na.rm = TRUE),
            .groups = "drop") %>%
  pivot_wider(names_from = arm,
              values_from = c(n, mean_conc, sd_conc),
              names_glue = "{arm}_{.value}") %>%
  # add p-values from t-test
  left_join(
    cyt_arm %>%
      group_by(cytokine) %>%
      summarise(p_value = t.test(conc ~ arm)$p.value, .groups = "drop"),
    by = "cytokine"
  ) %>%
  mutate(across(where(is.numeric), ~ round(., 3))) %>%
  kable(caption = "Cytokine concentrations by birth control arm (t-test)")

4. Regression Models

4.1 Linear regression: cytokine ~ arm + age + pcos_status

We model each cytokine as a function of birth control arm, age, and PCOS status.

cyt_reg <- data_cyt %>%
  pivot_longer(cols = all_of(cyt_names),
               names_to = "cytokine", values_to = "conc") %>%
  drop_na(conc, arm, age, pcos_status) %>%
  mutate(conc_log = log10(conc + 1))

models <- cyt_reg %>%
  group_by(cytokine) %>%
  nest() %>%
  mutate(
    model = map(data, ~ lm(conc_log ~ arm + age + pcos_status, data = .x)),
    tidied = map(model, tidy)
  ) %>%
  unnest(tidied) %>%
  select(cytokine, term, estimate, std.error, statistic, p.value) %>%
  mutate(across(where(is.numeric), ~ round(., 4)))
